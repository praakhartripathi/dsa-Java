class Solution {
    public int numSteps(String s) {
        int step = 0;
        int carry = 0;

        for (int i = s.length()-1 ; i > 0; i--) {
            int bit = (s.charAt(i) - '0') + carry;

            if (bit % 2 == 1) {
                step += 2;
                carry = 1;
            } else {
                step += 1;
            }
        }
        return step + carry;
    }
}

class Solution {
    public int numSteps(String s) {
        int n = Integer.parseInt(s, 2);
        int steps = 0;

        while (n > 1) {
            if (n % 2 == 1) {
                n = n + 1;
            } else {
                n = n / 2;
            }
            steps++;
        }
        return steps;
    }
}
