public int bitwiseComplement(int n) {
        // Example Dry Run: n = 5
        String binary = Integer.toBinaryString(n);
        // binary = "101"

        for (int i = 0; i < binary.length(); i++) {
            // i=0: charAt(0) is '1'. Else executes.
            //      binary becomes "" + '0' + "01" -> "001"
            // i=1: charAt(1) is '0' (in modified "001"). If executes.
            //      binary becomes "0" + '1' + "1" -> "011"
            // i=2: charAt(2) is '1' (in modified "011"). Else executes.
            //      binary becomes "01" + '0' + "" -> "010"
            if (binary.charAt(i) == '0') {
                binary = binary.substring(0, i) + '1' + binary.substring(i + 1);
            } else {
                binary = binary.substring(0, i) + '0' + binary.substring(i + 1);
            }
        }
        return Integer.parseInt(binary, 2);
        // Parsing "010" base 2 returns 2
    }
