class Solution {
    public int[] sortByBits(int[] arr) {
        int n = arr.length;

        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n -1; j++) {
                int b1 = countBits(arr[j]);
                int b2 = countBits(arr[j + 1]);

                if (b1 > b2 || (b1 == b2 && arr[j] > arr[j + 1])) {
                    int temp = arr[j];
                    arr[j] = arr[j + 1];
                    arr[j + 1] = temp;
                }
            }
        }
        return arr;
    }

    private String toBinary(int n) {
        if (n == 0) return "0";

        String binary = "";

        while (n > 0) {
            binary = (n % 2) + binary;
            n = n  / 2;
        }
        return binary;
    }

    private int countBits(int n) {
        String binary = toBinary(n);
        int count = 0;

        for (int i = 0; i < binary.length(); i++) {
            if (binary.charAt(i) == '1') {
                count++;
            }
        }
        return count;
    }
}
