class Solution {
    public int minOperations(String s) {
        int pattern1 = 0;
        int pattern2 = 0;

        for (int i = 0; i < s.length(); i++) {
            if (i % 2 == 0) {
                if (s.charAt(i) != '0') {
                    pattern1++;
                }
                if (s.charAt(i) != '1') {
                    pattern2++;
                }
            } else {
                if (s.charAt(i) != '1') {
                    pattern1++;
                }
                if (s.charAt(i) != '0') {
                    pattern2++;
                }
            }
        }
        return Math.min(pattern1, pattern2);
    }
}
