class Solution {
    public int minFlips(String s) {
        int n = s.length();
        String t = s + s;

        int mismatchA = 0;
        int mismatchB = 0;

        int left = 0;
        int ans = Integer.MAX_VALUE;

        for (int right = 0; right < 2 * n; right++) {
            char c = t.charAt(right);

            char expectedA = (right % 2 == 0) ? '0' : '1';
            char expectedB = (right % 2 == 0) ? '1' : '0';

            if (c != expectedA) {
                mismatchA++;
            }
            if (c != expectedB) {
                mismatchB++;
            }

            if (right - left + 1  > n) {
                char remove = t.charAt(left);
                
                char expA = (left % 2 == 0) ? '0' : '1';
                char expB = (left % 2 == 0) ? '1' : '0';

                if (remove != expA) {
                    mismatchA--;
                }

                if (remove != expB) {
                    mismatchB--;
                }

                left++;
            }

            if (right - left + 1 == n) {
                ans = Math.min(ans, Math.min(mismatchA, mismatchB));
            }
        }
        return ans;
    }
}
