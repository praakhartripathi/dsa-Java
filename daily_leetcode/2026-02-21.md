class Solution {
    public int countPrimeSetBits(int left, int right) {
        int totalPrimeCount = 0;

        for(int i = left; i <= right; i++) {
            int ones = countOnes(i);

            if (isPrime(ones)) {
                totalPrimeCount++;
            }
        }
        return totalPrimeCount++; 
    }

    public static boolean isPrime(int n) {
        int count = 0;

        for (int i = 1; i <= n; i++) {
            if (n % i == 0) {
                count++;
            }
        }
        if (count == 2) {
            return true;
        } else {
            return false;
        }
    }

    public static int countOnes(int n) {
        int count = 0;
        
        while (n > 0) {
            count += (n % 2);
            n /= 2;
        }
        return count;
    }
}

class Solution {
    public int countPrimeSetBits(int left, int right) {
        int count = 0;

        for (int i = left; i <= right; i++) {
            if (isSmallestPrime(Integer.bitCount(i))) {
                count++;
            }
        }
        return count;
    }

    public boolean isSmallestPrime(int i) {
        return (i == 2 || i == 3 || i == 5 || i == 7 || i == 11 || i == 13 || i == 17 || i == 19);
    }
}
