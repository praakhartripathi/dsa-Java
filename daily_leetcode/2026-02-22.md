class Solution {
    public int binaryGap(int n) {
        String binary = "";

        while (n > 0) {
            binary = (n % 2) + binary;
            n = n/2;
        }

        int lastIndex = -1;
        int maxGap = 0;

        for (int i = 0; i < binary.length(); i++) {
            if (binary.charAt(i) == '1') {
                if (lastIndex != -1) {
                    maxGap = Math.max(maxGap, i - lastIndex);
                }
            lastIndex = i;
            } 
        }
        return maxGap;
    }
}
