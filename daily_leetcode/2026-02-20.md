class Solution {
    public String makeLargestSpecial(String s) {
        List<String> list = new ArrayList<>();

        int count = 0;
        int start = 0;

        for (int i = 0; i < s.length(); i++) {
            if (s.charAt(i) == '1') {
                count++;
            } else {
                count--;
            }

            if (count == 0) {
                String inner = makeLargestSpecial(s.substring(start + 1, i));
                list.add("1" + inner + "0");
                start = i + 1;
            }
        }
        
        Collections.sort(list, Collections.reverseOrder());

            StringBuilder sb = new StringBuilder();
            for (String str : list) {
                sb.append(str);
            }
        return sb.toString();
    }
}

import java.util.*;

class Solution {

    public String makeLargestSpecial(String s) {

        String[] arr = new String[s.length()];
        int idx = 0;

        int count = 0;
        int start = 0;

        for (int i = 0; i < s.length(); i++) {

            if (s.charAt(i) == '1') count++;
            else count--;

            // balanced mountain mila
            if (count == 0) {

                String special = processManually(s, start, i);
                arr[idx++] = special;

                start = i + 1;
            }
        }

        for (int i = 0; i < idx; i++) {
            for (int j = i + 1; j < idx; j++) {
                if (arr[i].compareTo(arr[j]) < 0) {
                    String temp = arr[i];
                    arr[i] = arr[j];
                    arr[j] = temp;
                }
            }
        }

        String result = "";
        for (int i = 0; i < idx; i++) {
            result = result + arr[i];
        }

        return result;
    }

    private String processManually(String s, int start, int end) {

        String inner = "";

        for (int i = start + 1; i < end; i++) {
            inner = inner + s.charAt(i);
        }

        String processed = makeLargestSpecial(inner);

        return "1" + processed + "0";
    }
}
