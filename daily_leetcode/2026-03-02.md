class Solution {
    public int minSwaps(int[][] grid) {
        int n = grid.length;

        int[] endZeros = new int[n];

        for (int i = 0; i < n; i++) {
            int j = n - 1;

            int count = 0;
            while (j >= 0 && grid[i][j] == 0) {
                count++;
                j--;
            }

            endZeros[i] = count;
        }

        int step = 0;

        for (int i = 0; i < n; i++) {
            int need = n - i - 1;
            int j = i;

            while (j < n && endZeros[j] < need) {
                j++;
            }

            if (j == n) {
                return -1;
            }

            //step += n - 1;

            while (j > i) {
                int temp = endZeros[j];
                endZeros[j] = endZeros[j - 1];
                endZeros[j - 1] = temp;

                j--;
                step++;
            }
        }
        return step;
    }
}
